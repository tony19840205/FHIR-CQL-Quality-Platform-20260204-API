```python
# Python示例 - 集中式日期轉換
import json
from datetime import datetime, timedelta
import random

def convert_to_concentrated_dates(date_str, target_start_date="2026-01-01", days_range=5):
    """
    將日期轉換到集中的日期範圍內
    
    Args:
        date_str: 原始日期字串
        target_start_date: 目標開始日期 (2026-01-01 或 2026-04-01)
        days_range: 分散天數 (預設5天)
    """
    if not date_str:
        return date_str
    
    # 解析原始日期
    original_dt = datetime.fromisoformat(date_str.replace('Z', '+00:00'))
    
    # 解析目標開始日期
    target_start = datetime.fromisoformat(target_start_date + "T00:00:00+08:00")
    
    # 計算在原始月份中的位置 (0-1之間)
    day_position = (original_dt.day - 1) / 30  # 簡化：假設每月30天
    
    # 映射到目標日期範圍
    target_day_offset = int(day_position * days_range)
    target_dt = target_start + timedelta(days=target_day_offset)
    
    # 保持原始時間
    target_dt = target_dt.replace(
        hour=original_dt.hour,
        minute=original_dt.minute,
        second=original_dt.second
    )
    
    return target_dt.isoformat()

def convert_to_2026_q1(date_str):
    """轉換到 2026 Q1 (2026-01-01 ~ 2026-01-05)"""
    return convert_to_concentrated_dates(date_str, "2026-01-01", 5)

def convert_to_2026_q2(date_str):
    """轉換到 2026 Q2 (2026-04-01 ~ 2026-04-05)"""
    return convert_to_concentrated_dates(date_str, "2026-04-01", 5)
```
